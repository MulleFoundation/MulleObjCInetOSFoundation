if( NOT __MULLE_OBJC_OS_FOUNDATION_CMAKE_DEPENDENCIES_TXT__)
   set( __MULLE_OBJC_OS_FOUNDATION_CMAKE_DEPENDENCIES_TXT__ ON)

   include( ${DEPENDENCIES_DIR}/include/MulleObjCStandardFoundation/CMakeDependencies.txt)
   include( ${DEPENDENCIES_DIR}/include/MulleObjCInetFoundation/CMakeDependencies.txt)

   if( CMAKE_SYSTEM_NAME STREQUAL "Darwin")
      set( CMAKE_PLATFORM_FOUND true)

      set( HAVE_OSBASE ON)
      set( HAVE_POSIX ON)
      set( HAVE_BSD ON)
      set( HAVE_DARWIN ON)
   endif()


   if( CMAKE_SYSTEM_NAME STREQUAL "FreeBSD")
      set( CMAKE_PLATFORM_FOUND true)

      set( HAVE_OSBASE ON)
      set( HAVE_POSIX ON)
      set( HAVE_BSD ON)
      set( HAVE_FREEBSD ON)
   endif()

      set( CMAKE_PLATFORM_FOUND true)

      set( HAVE_OSBASE ON)
      set( HAVE_POSIX ON)
      set( HAVE_LINUX ON)
   endif()


   if( NOT CMAKE_PLATFORM_FOUND)
      message( STATUS "No support for ${CMAKE_SYSTEM_NAME}")
   else()
      message( STATUS "Building for platform ${CMAKE_SYSTEM_NAME}")
   endif()

   if( HAVE_OSBASE)
      if( NOT MULLE_OBJC_OSBASE_FOUNDATION_LIBRARY)
         find_library( MULLE_OBJC_OSBASE_FOUNDATION_LIBRARY NAMES MulleObjCOSBaseFoundation)
         message(STATUS "MULLE_OBJC_OSBASE_FOUNDATION_LIBRARY is ${MULLE_OBJC_OSBASE_FOUNDATION_LIBRARY}")
         if( MULLE_OBJC_OSBASE_FOUNDATION_LIBRARY)
            set( OBJC_DEPENDENCY_LIBRARIES
               ${MULLE_OBJC_OSBASE_FOUNDATION_LIBRARY}
               ${OBJC_DEPENDENCY_LIBRARIES}
               CACHE INTERNAL "need to cache this"
            )
            set( OBJC_DEPENDENCY_NAMES
               MulleObjCOSBaseFoundation
               ${OBJC_DEPENDENCY_NAMES}
               CACHE INTERNAL "need to cache this too"
            )
         endif()
      endif()
   endif()

   if( HAVE_POSIX)

               ${MULLE_OBJC_POSIX_FOUNDATION_LIBRARY}

               ${MULLE_OBJC_BSD_FOUNDATION_LIBRARY}

               CACHE INTERNAL "need to cache this too"
